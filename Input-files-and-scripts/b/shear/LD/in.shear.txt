label dloop
variable d loop 5


# 3d metal shear simulation
# -------
units                  metal 
dimension              3
boundary               p p s
atom_style             atomic 
timestep          0.001
read_data         LD.lmp
# ---------- Define Interatomic Potential --------------------- 
pair_style eam/alloy 
pair_coeff  * *  NiCoCr.lammps.eam    Ni  Co Cr
region        lower block INF INF INF INF INF 5 units box
region        upper block INF INF INF INF 118 INF units box
group        lower region lower
group        upper region upper
group        boundary union lower upper
group        mobile subtract all boundary

min_style       cg
minimize        1e-15 1e-15 10000 10000
reset_timestep    0

# Compute temperatures
compute        new3d mobile temp
compute        new2d mobile temp/partial 0 1 1
compute         vor all voronoi/atom
compute         peratom all stress/atom NULL

# equilibrate
velocity    mobile create 1.0 5812775 temp new3d
fix        1 all nve
fix        2 boundary setforce 0 0 0
fix        3 mobile temp/rescale 10 1 1 10 1.0
fix_modify    3 temp new3d
run        5000
reset_timestep    0
# shear
velocity    upper set 2 0 0 units box
velocity    mobile ramp vx 0.0 2 z 5 118 sum yes units box
fix        3 mobile temp/rescale 10 1 1 10 1
fix_modify    3 temp new2d

# Compute stress and volume
compute         pmob mobile reduce sum c_peratom[1] c_peratom[2] c_peratom[3] c_peratom[4] c_peratom[5] c_peratom[6]
compute         mobvol mobile reduce sum c_vor[1]


# Calculate average stress components
variable        prem1 equal c_pmob[1]/(c_mobvol*10000)                       #divided by 10000, GPa
variable        prem2 equal c_pmob[2]/(c_mobvol*10000)
variable        prem3 equal c_pmob[3]/(c_mobvol*10000)
variable        prem4 equal c_pmob[4]/(c_mobvol*10000)
variable        prem5 equal c_pmob[5]/(c_mobvol*10000)
variable        prem6 equal c_pmob[6]/(c_mobvol*10000)  

fix def all print 100 "${prem1} ${prem2} ${prem3} ${prem4} ${prem5} ${prem6}" file shear.mobile${d}.txt screen no

thermo 100
thermo_style custom step temp press pxx pyy pzz pxy pxz pyz

shell mkdir shear_process${d}
dump            1 all custom 100 shear_process${d}/dump.* id type x y z vx vy vz fx fy fz
dump_modify     1 sort id

run       10000

next d
clear
jump SELF dloop
